#define mem(a,b) memset(a, b, sizeof(a))

const int base = 31;

vector<ll>Pow, Hash, a;

ll getHash(int l, int r) {
return (Hash[r] - Hash[l - 1] * Pow[r - l + 1] % MOD + MOD) % MOD;
}
                
void __init__() {

    Pow[i] = (Pow[i - 1] * base) % MOD;
}
for(int i = 1; i <= n; i++) {
    Hash[i] = (Hash[i - 1] * base + (s[i] - 'a' + 1)) % MOD;
    }
}

void solve() {

    __init__();
    for(int k = 1; k <= n; k++) {
        ll pre = getHash(1, k);
        ll suf = getHash(n - k + 1, n);
   //     cout << pre << ' ' << suf << endl;

        }

        cout << "Just a legend\n";
    } else {
        int len = -1;

            for(int i = 2; i <= n - k; i++) {

                    cout << s.substr(1, k) << endl; exit(0);
                }
            }

      }

#include <bits/stdc++.h>

#define mem(a,b) memset(a, b, sizeof(a))

#define faster() ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
const int N = 2e5 + 15;

vector<int>dp(N, 0);

int digit(int n) {
	int sum = 0;
	while(n) {
		sum = sum + (n % 10);
		n = n / 10; 
	}
	return sum; 
} 

void __init__() {
	dp[1] = 1;
	for(int i = 2; i <= N; i++) {
		dp[i] = dp[i - 1] + digit(i); 
	} 
} 

void solve() {
	int n; cin >> n;
	cout << dp[n] << endl; 
}

    faster();
    __init__();
    int t; cin >> t;
    while(t--) {
        solve();
    }
